Implementación TDD para Incrementar Stock

Criterio de Aceptación

El sistema debe permitir incrementar el stock de un producto existente

Fase Verde: Implementación que pasa las pruebas

ProductService (Implementación completa)

package com.store.domain.service;

import com.store.domain.port.ProductRepository;
import com.store.domain.model.Product;

public class ProductService {
    private final ProductRepository productRepository;

    public ProductService(ProductRepository productRepository){
        this.productRepository = productRepository;
    }

    public void incrementStock(String productName, int increment) {
        // Validar que el incremento no sea negativo
        if (increment < 0) {
            throw new IllegalArgumentException("El incremento no puede ser negativo");
        }
        
        // Buscar el producto
        Product product = productRepository.findByName(productName);
        if (product == null) {
            throw new IllegalArgumentException("Producto no encontrado");
        }
        
        // Incrementar el stock y guardar
        product.setStock(product.getStock() + increment);
        productRepository.save(product);
    }

    // Métodos existentes...
    public void saveProduct(Product product) {
        if (product.getStock() < 0){
            throw new IllegalArgumentException("El stock no debe ser negativo.");
        }
        productRepository.save(product);
    }

    public void setMinimumStockLevel(Product product, int minimumStockLevel){
        validateMinimumStockLevel(minimumStockLevel);
        product.setMinimumStockLevel(minimumStockLevel);
        saveProduct(product);
    }

    private void validateMinimumStockLevel(int minimumStockLevel){
        if (minimumStockLevel <= 0){
            throw new IllegalArgumentException("El nivel mínimo debe ser mayor a cero.");
        }
    }
}
